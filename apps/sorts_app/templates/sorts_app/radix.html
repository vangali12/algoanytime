[Radix Sort]

Input: Unsorted Array

Output: Sorted Array

Begin by sorting all of the values by the one's place. Then keeping the order, combine the values back into a single array. Continue by sorting the values by the tens, hundreds, until you sort by the number of places of the largest value of your set. Your final combined array is your ordered array.

function radixSort(arr) {
    function findMaxLen(col) {
        let max = col[0];
        for (var i=0; i max) {
                max = col[i];
            }
        }
        return max;
    }
    maxLength = findMaxLen(arr).toString().length;
    console.log(maxLength)
    var newArr = arr;
    var sortedArr = [];
    for (var i = 0; i < maxLength; i++) {
        for (var j = 0; j<10; j++) {
            sortedArr[j] = [];
        }
        for (var k = 0; k < newArr.length; k++) {
            sortedArr[Math.floor(newArr[k]/Math.pow(10,i))%10].push(newArr[k]);
        }
        newArr=[];
        for (var m = 0; m < sortedArr.length; m++) {
            newArr = newArr.concat(sortedArr[m]);
        }
    }
    return newArr;
}
