BST.prototype.delete = function(num) {
	if(!this.root) {
		return false;
	} else if (!this.contains(num)) {
		return false;
	}
	var runner = this.root;
	var checker = this.root;
	while (checker.val != num) {
		runner = checker;
		if (num > runner.val) {
			checker = runner.right;
		} else if (num < runner.val) {
			checker = runner.left;
		}
	}
	if (!checker.right && !checker.left) {
		if (runner.val>checker.val) {
			runner.left = null;
			return this;
		} else {
			runner.right = null;
			return this;
		}
	} else if (!checker.left && checker.right) {
		if (runner.val>checker.val) {
			runner.left = checker.right;
			return this;
		} else {
			runner.right = checker.right;
			return this;
		}
	} else if (!checker.right && checker.left) {
		if (runner.val>checker.val) {
			runner.left = checker.left;
			return this;
		} else {
			runner.right = checker.left;
			return this;
		}
	} else {
		var cur = checker.right;
		while (cur.left) {
			temp = cur;
			cur = cur.left;
		}
		temp.left = null;
		var replacement = new BTnode(cur.val);
		replacement.right = checker.right;
		replacement.left = checker.left;
		if (runner.val > checker.val) {
			runner.left = replacement;
			return this;
		} else {
			runner.right = replacement;
			return this;
		}
	}
}
