function repeated(num) {
	var str = num.toString()
	var dict = {};
	var ans = {};
	function helper(count) {
		if (count == (num.toString().length-1)){
			val = str[count];
			if (dict[val]) {
				dict[val] += 1
			} else {
				dict[val] = 1;
			}
			for (val in dict) {
				if (dict[val] > 1) {
					ans[val] = dict[val];
				}	
			}
			return ans;
		}
		val = str[count];
		if (dict[val]) {
			dict[val] += 1
		} else {
			dict[val] = 1;
		}
		return helper(count + 1);
	}
	return helper(0)
}
