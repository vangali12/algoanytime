function qTree(str) {
	var arr = [];
	function findQF(str) {
		var front = "";
		for (var i = 0; i < str.length; i++) {
			if (str[i] == "?") {
				return front;
			} else {
				front += str[i]
			}
		}
		return str;
	}
	function findQB(str, num) {
		console.log(str)
		console.log(num)
		var back = num.toString();
		for (var j = 0; j < str.length; j++) {
			if (str[j] == "?") {
				break;
			}
		}
		for (var k = j+1; k < str.length; k++) {
			back += str[k];
		}
		return back;
	}
	function replace(f, b) {
		console.log(f)
		console.log(b)
		if (f.length == str.length) {
			console.log("f at bc: " + f)
			console.log("b at bc: " + b)
			arr.push(f);
			console.log("array at bc: " + arr)
			f = "";
			b = "";
			return;
		}
		replace(findQF(f+b), findQB((f+b), 0));
		replace(findQF(f+b), findQB((f+b), 1));
	}
	replace(findQF(str), findQB(str, 0));
	replace(findQF(str), findQB(str, 1));
	return arr;
}
