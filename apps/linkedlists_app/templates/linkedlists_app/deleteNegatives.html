Sll.prototype.delNeg = function() {
	if(!this.head) {
		return false;
	} else if (!this.head.next){
		if(this.head.val < 0) {
			this.head = null;
		}
		return self;
	} else {
		var runner = this.head;
		var checker = this.head.next;
		while(checker.next) {
			if (!checker.val) {
				return self;
			} else if (checker.val < 0) {
				checker = checker.next
			}
			runner.next = checker;
			runner = checker;
			checker = checker.next;
		}
	}
	if (this.head.val < 0) {
		this.removeFront();
	}
	return self;
}
